
//2 tane ADC çekirdeği kullanarak 1Mhz frekansında Okuma yapar                ADC pinleri 24 ve 28
//ve bunları interrupt aracılığıyla DACA ve DACB'den çıkartır  				  DAC pinleri 30 ve 70

//NOT: 2 tane çekirdek kullanmasının sebebi ölçümlerin doğruluğunun artması ve daha yüksek frekanslara izin vermesidir

#include "f28x_project.h"

void initPWM1(void);
void initADCB(void);
void initADCC(void);
void initDACA(void);
void initDACB(void);

__interrupt void adcb1_isr(void);

void main(void){
  InitSysCtrl();

  DINT;

  EALLOW;
  IER = 0x0000;
  IFR = 0x0000;
  EDIS;

  InitPieCtrl();
  InitPieVectTable();

  EALLOW;
  PieVectTable.ADCB1_INT = &adcb1_isr;  //ADCB1 interrupt oluşturduğunda PIE modülüne ben nereye gidecem diye soran 
  EDIS;                                 //CPU'ya gösterilecek adresi seç(interrupt fonksiyonumuz)
 


  EALLOW;
  PieCtrlRegs.PIEIER1.bit.INTx2 = 1;    //2. grubun 1. üyesinin vektör tablosunun kullanılmasına izin ver
  EDIS;

  EALLOW;
  IER |= M_INT1;    //1. grubun interrupt'larına izin ver(ADC için)
  EDIS;


  initPWM1();
  initDACA();
  initDACB();
  initADCB();
  initADCC();

  EINT;

  for(;;);
}

void adcb1_isr(void){
  //ADC ölçümlerini DAC'lara girmek
  DacaRegs.DACVALS.bit.DACVALS = AdcbResultRegs.ADCRESULT0;
  DacbRegs.DACVALS.bit.DACVALS = AdccResultRegs.ADCRESULT0;

  //Bayrak temizliği
  EALLOW;
  AdcbRegs.ADCINTOVFCLR.bit.ADCINT1 = 1;
  AdcbRegs.ADCINTFLGCLR.bit.ADCINT1 = 1;
  PieCtrlRegs.PIEACK.bit.ACK1 = 1;
  EDIS;
}

void initPWM1(void){
EALLOW;
//1MHz PWM çalışması için gereken ayarlar
EPwm1Regs.TBCTL.bit.FREE_SOFT = 0;      //Free modda çalıştır
EPwm1Regs.TBCTL.bit.CTRMODE = 0;        //Up count mode
EPwm1Regs.TBCTL.bit.HSPCLKDIV = 0;      //SYSCLKOUT frekansını bölme => /1
EPwm1Regs.TBCTL.bit.PRDLD = 0;          //SHADOW register kullan

EPwm1Regs.TBPRD = 100;                 //1MHz çalış(kart maksimum 3.5MHz civarında okuyabiliyor)

EPwm1Regs.CMPCTL.bit.SHDWAMODE = 0;     //CMPA için shadow mode açtık
EPwm1Regs.CMPCTL.bit.LOADAMODE = 0;     //TBCLK 0'landığını CMPA'yı tekrar yükle

EPwm1Regs.CMPA.bit.CMPA = 50;          //Duty cycle %50 olması için cmpa periyodun yarısı yaptık

EPwm1Regs.AQCTLA.bit.ZRO = 0;           //TBCLK sıfırlandığında hiçbir şey yapma
EPwm1Regs.AQCTLA.bit.CAU = 0;           //CMPA yukarı sayarken TBCLK'ya eşit olursa hiçbir şey yapma

//ADC'ye SOCA başlatmak için gereken ayarlar
EPwm1Regs.ETSEL.bit.SOCAEN = 0;         //SOCA oluşturmaya izin VERME
EPwm1Regs.ETSEL.bit.SOCASEL = 2;        //TBCLK TBPRD'ye eşit olduğunda event(SOCA start sinyali) oluştur
EPwm1Regs.ETPS.bit.SOCAPRD = 1;         //Her SOCA üretme isteğinde SOC üret
EPwm1Regs.ETSEL.bit.SOCAEN = 1;         //SOCA oluşturmaya izin VER
EDIS;
}



void initADCB(void){
SetVREF(ADC_ADCB, ADC_INTERNAL,0);      //ADC çalışması için gerekli ön ayarları yapar
EALLOW;
AdcbRegs.ADCCTL1.bit.ADCPWDNZ = 1;      //ADC'ye enerji gitmesini sağla
DELAY_US(1000);                         //ADC  çalışmaya başlayana kadar bekle
AdcbRegs.ADCCTL2.bit.PRESCALE = 2;      //SYSCLKOUT frekansını 4'ye böler

AdcbRegs.ADCSOC0CTL.bit.CHSEL = 0;      //ADCIN0 tetiklensin
AdcbRegs.ADCSOC0CTL.bit.ACQPS = 7;      //7 saat döngüsü boyunca kapasitörü doldur(kesinlikle yeterli denendi)
AdcbRegs.ADCSOC0CTL.bit.TRIGSEL = 5;    //ePWM1, ADCSOCA tarafından tetiklensin

AdcbRegs.ADCSOCPRICTL.bit.SOCPRIORITY = 2; //SOC0 high priority modunda, diğerleri round robinde

//Interrupt ayarları
AdcbRegs.ADCCTL1.bit.INTPULSEPOS = 1;
AdcbRegs.ADCINTSEL1N2.bit.INT1E = 1;
AdcbRegs.ADCINTSEL1N2.bit.INT1CONT = 0;
AdcbRegs.ADCINTSEL1N2.bit.INT1SEL = 0;
EDIS;
}

void initADCC(void){
SetVREF(ADC_ADCC, ADC_INTERNAL,0);      //ADC çalışması için gerekli ön ayarları yapar
EALLOW;
AdccRegs.ADCCTL1.bit.ADCPWDNZ = 1;      //ADC'ye enerji gitmesini sağla
DELAY_US(1000);                         //ADC  çalışmaya başlayana kadar bekle
AdccRegs.ADCCTL2.bit.PRESCALE = 2;      //SYSCLKOUT frekansını 4'ye böler

AdccRegs.ADCSOC0CTL.bit.CHSEL = 0;      //ADCIN0 tetiklensin
AdccRegs.ADCSOC0CTL.bit.ACQPS = 7;      //7 saat döngüsü boyunca kapasitörü doldur(kesinlikle yeterli denendi)
AdccRegs.ADCSOC0CTL.bit.TRIGSEL = 5;    //ePWM1, ADCSOCA tarafından tetiklensin

AdccRegs.ADCSOCPRICTL.bit.SOCPRIORITY = 1; //SOC0 high priority modunda, diğerleri round robinde
EDIS;
}



  void initDACA(void){
  EALLOW;
  DacaRegs.DACCTL.bit.DACREFSEL = 1;
  DacaRegs.DACCTL.bit.LOADMODE = 0;
  DacaRegs.DACOUTEN.bit.DACOUTEN = 1;
  DacaRegs.DACCTL.bit.MODE = 1;

  AnalogSubsysRegs.ANAREFCTL.bit.ANAREFASEL = 0;
  AnalogSubsysRegs.ANAREFCTL.bit.ANAREFA2P5SEL = 0;
  EDIS;
  }

  void initDACB(void){
  EALLOW;
  DacbRegs.DACCTL.bit.DACREFSEL = 1;
  DacbRegs.DACCTL.bit.LOADMODE = 0;
  DacbRegs.DACOUTEN.bit.DACOUTEN = 1;
  DacbRegs.DACCTL.bit.MODE = 1;

  AnalogSubsysRegs.ANAREFCTL.bit.ANAREFBSEL = 0;
  AnalogSubsysRegs.ANAREFCTL.bit.ANAREFB2P5SEL = 0;
  EDIS;
  }

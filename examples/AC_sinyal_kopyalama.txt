void main(void){

    InitSysCtrl();

    DINT;

    InitGpio();

    //GPIO Ayarları
    EALLOW;
    GpioCtrlRegs.GPAMUX1.bit.GPIO0 = 1;
    GpioCtrlRegs.GPAPUD.bit.GPIO0 = 1;

    GpioCtrlRegs.GPAMUX1.bit.GPIO1 = 1;
    GpioCtrlRegs.GPAPUD.bit.GPIO1 = 1;
    
    GpioCtrlRegs.GPAMUX1.bit.GPIO2 = 1;
    GpioCtrlRegs.GPAPUD.bit.GPIO2 = 1;

    GpioCtrlRegs.GPAMUX1.bit.GPIO3 = 1;
    GpioCtrlRegs.GPAPUD.bit.GPIO3 = 1;
    EDIS;

    EALLOW;
    IER = 0x0000;
    IFR = 0x0000;
    EDIS;
    
    sinwave = RMSSIN[i];
    
    InitPieCtrl();
    InitPieVectTable();

    EALLOW;
    PieVectTable.ADCA1_INT = &adca1_isr;
    EDIS;

    EALLOW;
    PieCtrlRegs.PIEIER1.bit.INTx1 = 1;
    EDIS;

    EALLOW;
    IER |= M_INT1;    //1. grubun interrupt'larına izin ver(ADC için)
    EDIS;

    EINT;

    initPWM1();
    initPWM2();
    initPWM3();
    initDACA();
    initADCA();

    for(;;);
}

void adca1_isr(void){
    //ADC ölçümlerini DAC'lara girmek
    //DacaRegs.DACVALS.bit.DACVALS = AdcbResultRegs.ADCRESULT0;
    if( !start & AdcaResultRegs.ADCRESULT0 > 3500){
        EPwm1Regs.TBCTL.bit.SWFSYNC = 1;
        start = true;
    }

    DacaRegs.DACVALS.bit.DACVALS = sinwave;

    if(start){
        if(divider){
            divider = false;
            sinwave = RMSSIN[i];

            // 2. Indisi guncelle
            if (yon == false) // Azaliyor
            {
                if (i > 0) {
                    i--;
                } else {
                    yon = true; // Yon degistir: Artik artacak
                    i++;        // 0'dan 1'e cik
                }
            }
            else // Artiyor
            {
                if (i < 998) {
                    i++;
                } else {
                    yon = false; // Yon degistir: Artik azalacak
                    i--;         // 999'dan 998'e in
                }
            }
        }
        else {
            divider = true;
            }
    }

    //Bayrak temizliği
    EALLOW;
    AdcaRegs.ADCINTOVFCLR.bit.ADCINT1 = 1;
    AdcaRegs.ADCINTFLGCLR.bit.ADCINT1 = 1;
    PieCtrlRegs.PIEACK.bit.ACK1 = 1;
    EDIS;
}

void initPWM1(void){
EALLOW;
//100hz PWM çalışması için gereken ayarlar
EPwm1Regs.TBCTL.bit.FREE_SOFT = 0;      //Soft modda çalıştır
EPwm1Regs.TBCTL.bit.CTRMODE = 0;        //Up count mode
EPwm1Regs.TBCTL.bit.HSPCLKDIV = 0;      //SYSCLKOUT frekansını bölme => /1
EPwm1Regs.TBCTL.bit.PRDLD = 0;          //SHADOW register kullan
EPwm1Regs.TBCTL.bit.CLKDIV = 6;         //SYSCLKOUT frekansını 64'e böl => /64

EPwm1Regs.TBPRD = 15624 * 2;                //Periyot 15625 ayarlandı. 15624*6 = 1.000.000 yani 100hz ayarlamış olduk

EPwm1Regs.CMPCTL.bit.SHDWAMODE = 0;     //CMPA için shadow mode açtık
EPwm1Regs.CMPCTL.bit.LOADAMODE = 0;     //TBCLK 0'landığında CMPA'yı tekrar yükle

EPwm1Regs.CMPA.bit.CMPA = 7813*2;         //Duty cycle %50 olması için cmpa periyodun yarısı yaptık
EPwm1Regs.CMPB.bit.CMPB = 7813*2;         //Duty cycle %50 olması için cmpa periyodun yarısı yaptık

EPwm1Regs.AQCTLA.bit.ZRO = 2;           //0=OFF, 1=LOW, 2=HIGH, 3=TOGGLE
EPwm1Regs.AQCTLA.bit.CAU = 1;           //0=OFF, 1=LOW, 2=HIGH, 3=TOGGLE

EPwm1Regs.AQCTLB.bit.ZRO = 2;           //0=OFF, 1=LOW, 2=HIGH, 3=TOGGLE
EPwm1Regs.AQCTLB.bit.CBU = 1;           //0=OFF, 1=LOW, 2=HIGH, 3=TOGGLE


//DBD ayarları
EPwm1Regs.DBCTL.bit.OUT_MODE = 3;       //Hem RED hem de FED açık
EPwm1Regs.DBCTL.bit.IN_MODE = 2;        //EPWM2A'yı hem RED hem de FED için kullan
EPwm1Regs.DBCTL.bit.POLSEL = 2;         //ePWMB invert edilecek
EPwm1Regs.DBRED.bit.DBRED = 1;          //1 saat döngüsü bekle
EPwm1Regs.DBFED.bit.DBFED = 1;          //1 saat döngüsü bekle

EPwm1Regs.TBCTL.bit.PHSEN = 0;
EPwm1Regs.TBCTL.bit.SYNCOSEL = 1;
EDIS;

}

void initPWM3(void){
EALLOW;
//1MHz PWM çalışması için gereken ayarlar
EPwm3Regs.TBCTL.bit.FREE_SOFT = 0;      //Free modda çalıştır
EPwm3Regs.TBCTL.bit.CTRMODE = 0;        //Up count mode
EPwm3Regs.TBCTL.bit.HSPCLKDIV = 0;      //SYSCLKOUT frekansını bölme => /1
EPwm3Regs.TBCTL.bit.PRDLD = 0;          //SHADOW register kullan

EPwm3Regs.TBPRD = 500;                 //2MHz çalış(kart maksimum 3.5MHz civarında okuyabiliyor)

EPwm3Regs.AQCTLA.bit.CAU = 0;           //CMPA yukarı sayarken TBCLK'ya eşit olursa hiçbir şey yapma

//ADC'ye SOCA başlatmak için gereken ayarlar

EPwm3Regs.ETSEL.bit.SOCAEN = 1;         //SOCA oluşturmaya izin VER
EDIS;
}



void initADCA(void){

AdcaRegs.ADCCTL1.bit.ADCPWDNZ = 1;      //ADC'ye enerji gitmesini sağla

AdcaRegs.ADCCTL2.bit.PRESCALE = 0;      //SYSCLKOUT frekansını 2'ye böler

AdcaRegs.ADCSOC0CTL.bit.CHSEL = 1;      //ADCIN0 tetiklensin
AdcaRegs.ADCSOC0CTL.bit.ACQPS = 10;      //7 saat döngüsü boyunca kapasitörü doldur(kesinlikle yeterli denendi)
AdcaRegs.ADCSOC0CTL.bit.TRIGSEL = 9;    //ePWM1, ADCSOCA tarafından tetiklensin

AdcaRegs.ADCSOCPRICTL.bit.SOCPRIORITY = 0; //SOC0 high priority modunda, diğerleri round robinde

//Interrupt ayarları
AdcaRegs.ADCCTL1.bit.INTPULSEPOS = 1;
AdcaRegs.ADCINTSEL1N2.bit.INT1E = 1;
AdcaRegs.ADCINTSEL1N2.bit.INT1CONT = 0;
AdcaRegs.ADCINTSEL1N2.bit.INT1SEL = 0;
EDIS;
}

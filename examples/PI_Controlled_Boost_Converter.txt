#include "f28x_project.h"

//fonksiyon ön tanımlamaları
void initPWM1(void);
void initPWM2(void);
void initADCA(void);
void initDACA(void);
__interrupt void adca1_isr(void);

// Global Değişkenler
float adc_gerilim, gercek_gerilim;
float istenilen_cikis_gerilimi = 20.0;

// PI Kontrol Parametreleri
float Kp = 0.5;           // Proportional katsayısı (Deneyerek optimize edilmeli)
float Ki = 0.00001;         // Integral katsayısı
float integral_sum = 0.0; // Hata birikimi (Integral terimi)

float hata = 0.0;
float pi_cikti = 0.0;

void main(void){

    DINT;
    
    InitSysCtrl();

    InitGpio();

    InitPieCtrl();
    InitPieVectTable();

    EALLOW;
    PieVectTable.ADCA1_INT = &adca1_isr;   
    EDIS;                                 
    
    EALLOW;
    PieCtrlRegs.PIEIER1.bit.INTx1 = 1;    //1. grubun 1. üyesinin vektör tablosunun kullanılmasına izin ver
    EDIS;

    EALLOW;
    IER = 0x0000;
    IFR = 0x0000;
    EDIS;

    EALLOW;
    IER |= M_INT1;    //1. grubun interrupt'larına izin ver(ADC için)
    EDIS;

    //GPIO Ayarları
    EALLOW;
    GpioCtrlRegs.GPAMUX1.bit.GPIO0 = 1;
    GpioCtrlRegs.GPAPUD.bit.GPIO0 = 1;

    GpioCtrlRegs.GPAMUX1.bit.GPIO1 = 1;
    GpioCtrlRegs.GPAPUD.bit.GPIO1 = 1;
  
    EDIS;

    initPWM1();
    initPWM2();
    initADCA();
    initDACA();

    EINT;
    for(;;);
}


void initPWM1(void){
EALLOW;

EPwm1Regs.TBCTL.bit.FREE_SOFT = 0;      //free modda çalıştır
EPwm1Regs.TBCTL.bit.CTRMODE = 0;        //Up count mode
EPwm1Regs.TBCTL.bit.HSPCLKDIV = 0;      //SYSCLKOUT frekansını bölme => /1
EPwm1Regs.TBCTL.bit.PRDLD = 0;          //SHADOW register kullan
EPwm1Regs.TBCTL.bit.CLKDIV = 0;         //SYSCLKOUT frekansını 1'e böl

EPwm1Regs.TBPRD = 1000;                //Periyot 1000 ayarlandı. 1000 = 1.000.000 yani 100KHz ayarlamış olduk

EPwm1Regs.CMPCTL.bit.SHDWAMODE = 0;     //CMPA için shadow mode açtık
EPwm1Regs.CMPCTL.bit.LOADAMODE = 0;     //TBCLK 0'landığında CMPA'yı tekrar yükle

EPwm1Regs.CMPA.bit.CMPA = 500;         //Duty cycle %50 olması için cmpa periyodun yarısı yaptık
EPwm1Regs.CMPB.bit.CMPB = 500;         //Duty cycle %50 olması için cmpa periyodun yarısı yaptık

EPwm1Regs.AQCTLA.bit.ZRO = 2;           //0=OFF, 1=LOW, 2=HIGH, 3=TOGGLE
EPwm1Regs.AQCTLA.bit.CAU = 1;           //0=OFF, 1=LOW, 2=HIGH, 3=TOGGLE

EPwm1Regs.AQCTLB.bit.ZRO = 2;           //0=OFF, 1=LOW, 2=HIGH, 3=TOGGLE
EPwm1Regs.AQCTLB.bit.CBU = 1;           //0=OFF, 1=LOW, 2=HIGH, 3=TOGGLE


//DBD ayarları
EPwm1Regs.DBCTL.bit.OUT_MODE = 3;       //Hem RED hem de FED açık
EPwm1Regs.DBCTL.bit.IN_MODE = 2;        //EPWM2A'yı hem RED hem de FED için kullan
EPwm1Regs.DBCTL.bit.POLSEL = 2;         //ePWMB invert edilecek
EPwm1Regs.DBRED.bit.DBRED = 50;         //1 saat döngüsü bekle
EPwm1Regs.DBFED.bit.DBFED = 50;         //1 saat döngüsü bekle

EDIS;

}
void initPWM2(void){
EALLOW;
//1MHz PWM çalışması için gereken ayarlar
EPwm2Regs.TBCTL.bit.FREE_SOFT = 0;      //Free modda çalıştır
EPwm2Regs.TBCTL.bit.CTRMODE = 0;        //Up count mode
EPwm2Regs.TBCTL.bit.HSPCLKDIV = 0;      //SYSCLKOUT frekansını bölme => /1
EPwm2Regs.TBCTL.bit.PRDLD = 0;          //SHADOW register kullan

EPwm2Regs.TBPRD = 100;                 //1MHz çalış(kart maksimum 3.5MHz civarında okuyabiliyor)

EPwm2Regs.CMPCTL.bit.SHDWAMODE = 0;     //CMPA için shadow mode açtık
EPwm2Regs.CMPCTL.bit.LOADAMODE = 0;     //TBCLK 0'landığını CMPA'yı tekrar yükle

EPwm2Regs.CMPA.bit.CMPA = 50;          //Duty cycle %50 olması için cmpa periyodun yarısı yaptık

EPwm2Regs.AQCTLA.bit.ZRO = 0;           //TBCLK sıfırlandığında hiçbir şey yapma
EPwm2Regs.AQCTLA.bit.CAU = 0;           //CMPA yukarı sayarken TBCLK'ya eşit olursa hiçbir şey yapma

//ADC'ye SOCA başlatmak için gereken ayarlar
EPwm2Regs.ETSEL.bit.SOCAEN = 0;         //SOCA oluşturmaya izin VERME
EPwm2Regs.ETSEL.bit.SOCASEL = 2;        //TBCLK TBPRD'ye eşit olduğunda event(SOCA start sinyali) oluştur
EPwm2Regs.ETPS.bit.SOCAPRD = 1;         //Her SOCA üretme isteğinde SOC üret
EPwm2Regs.ETSEL.bit.SOCAEN = 1;         //SOCA oluşturmaya izin VER
EDIS;
}

void initADCA(void){
SetVREF(ADC_ADCA, ADC_INTERNAL,0);      //ADC çalışması için gerekli ön ayarları yapar
EALLOW;
AdcaRegs.ADCCTL1.bit.ADCPWDNZ = 1;      //ADC'ye enerji gitmesini sağla
DELAY_US(1000);                         //ADC  çalışmaya başlayana kadar bekle
AdcaRegs.ADCCTL2.bit.PRESCALE = 1;      //SYSCLKOUT frekansını 2'ye böler

AdcaRegs.ADCSOC0CTL.bit.CHSEL = 1;      //ADCIN1 tetiklensin
AdcaRegs.ADCSOC0CTL.bit.ACQPS = 7;      //7 saat döngüsü boyunca kapasitörü doldur(kesinlikle yeterli denendi)
AdcaRegs.ADCSOC0CTL.bit.TRIGSEL = 7;    //ePWM2, ADCSOCA tarafından tetiklensin

AdcaRegs.ADCSOCPRICTL.bit.SOCPRIORITY = 1; //SOC0 ve SOC1 high priority modunda, diğerleri round robinde
//İnterrupt iicn ayarlar
AdcaRegs.ADCCTL1.bit.INTPULSEPOS = 1;
AdcaRegs.ADCINTSEL1N2.bit.INT1E = 1;
AdcaRegs.ADCINTSEL1N2.bit.INT1CONT = 0;
AdcaRegs.ADCINTSEL1N2.bit.INT1SEL = 0;

EDIS;
}
  void initDACA(void){
  EALLOW;
  DacaRegs.DACCTL.bit.DACREFSEL = 1;
  DacaRegs.DACCTL.bit.LOADMODE = 0;
  DacaRegs.DACOUTEN.bit.DACOUTEN = 1;
  DacaRegs.DACCTL.bit.MODE = 1;

  AnalogSubsysRegs.ANAREFCTL.bit.ANAREFASEL = 0;
  AnalogSubsysRegs.ANAREFCTL.bit.ANAREFA2P5SEL = 0;
  EDIS;
  }

  
__interrupt void adca1_isr(void){
    // 1. Geri besleme (Feedback) oku ve ölçeklendir
    adc_gerilim = (AdcaResultRegs.ADCRESULT0 * 3.3) / 4096.0;
    gercek_gerilim = adc_gerilim * 10.0; // Gerilim bölücü oranına göre (3.3V ADC = 33V Çıkış)

    // 2. Hata Hesapla
    hata = istenilen_cikis_gerilimi - gercek_gerilim;

    // 3. Integral Terimi Hesapla (Hata Toplama)
    integral_sum += hata;

    // --- ANTI-WINDUP (Integral Sınırlama) ---
    // Integral teriminin sonsuza gitmesini önlemek için PWM limitlerine göre kısıtlıyoruz
    if (integral_sum > 0.8) integral_sum = 0.8; 
    if (integral_sum < 0.1) integral_sum = 0.1;

    // 4. PI Çıktısını Hesapla
    // u(t) = Kp*e(t) + Ki*integral(e)
    pi_cikti = (Kp * hata) + (Ki * integral_sum);

    // 5. Duty Cycle Sınırlama (Saturation)
    // TBPRD = 1000 olduğu için çıktı 0 ile 1000 arasında olmalı.
    // Boost converter için %80'den fazla duty tehlikelidir, 200-800 arası güvenlidir.
    float duty_cycle = pi_cikti * 1000.0; // Normalize değeri periyoda çevir

    if (duty_cycle > 800.0) duty_cycle = 800.0;
    if (duty_cycle < 200.0) duty_cycle = 200.0;

    // 6. PWM Register Güncelleme
    // Senkron yapıda CMPA anahtarlamayı yönetir, dead-band modülü diğerini halleder
    EPwm1Regs.CMPA.bit.CMPA = (uint16_t)duty_cycle;
    //DacbRegs.DACVALS.bit.DACVALS = 4*EPwm1Regs.CMPA.bit.CMPA;

    // Bayrak temizliği
    AdcaRegs.ADCINTFLGCLR.bit.ADCINT1 = 1;
    PieCtrlRegs.PIEACK.bit.ACK1 = 1;

}
